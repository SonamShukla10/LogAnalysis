import psycopg2
DBNAME = "popular"
def run(query):
"""Connects to the database, runs the query passes to it,
and returns the rresults"""
db = psycopg2.connect('dbname=' + DBNAME)
c = db.cursor()
c.execute(query)
rows=c.fetchall()
db.close()
return rows
def top_Articles() :
"""Returns top 3 most read article"""
# Build Query String
query = """
SELECT articles.title, COUNT(*) AS num FROM articles
join log
ON log.path LIKE concat('/article/%', articles.slug)
GROUP BY articles.title
ORDER BY articles.title
ORDER BY num DESC
LIMIT 3;
"""
# run #########
results = run(query)
print('\nTop Three Articles By Page views:')
count = 1
for i in results :
number = '(' + str(count) + ') " '
title = i[0]
views = '"With ' + str(i[1]) + "views"
print(number + title + views)
count +=1
def top_article_authors() :
"""returns top 3 most popular authors"""
# query string
query = """
SELECT authors.name, COUNT(*) AS num
FROM authors
Join articles
ON authors.id = articles.author
JOIN log
ON log.path like concat('/article/%, articles.slug)
GROUP BY authors.name
ORDER BY num DESC
LIMIT 3;
"""
# run########
results = run(query)
print('\nTOP THREE AUTHORS BY VIEWS :')
count = 1
for i in results:
print('(' + str(count) + ') ' + i[0] + ' with ' + str(i[1]) + "views")
count += 1
